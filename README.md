```bash
git clone https://github.com/teamparsinta/vite-inertia.git
cd vite-inertia
composer install
cp .env.example .env
php artisan key:generate
npm install
php artisan serve
```
Open new tab on terminal and run the vite with
```bash
npm run dev
```
To make it production run `npm run build`
